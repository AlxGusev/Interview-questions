Task 1:
SELECT
    MAX(salary) AS second_highest_salary
FROM employees
WHERE salary < (SELECT MAX(salary) FROM employees);

Task 2:
SELECT
       e.name AS employee_name
     , d.name AS department_name
     , e.salary, RANK() OVER (PARTITION BY e.department_id ORDER BY e.salary DESC) AS salary_rank
     , ROUND(e.salary - AVG(e.salary) OVER (PARTITION BY e.department_id), 2) AS diff_from_dept_avg
FROM employees e
LEFT JOIN departments d ON e.department_id = d.id
ORDER BY department_name, salary_rank;

Task 3:
SELECT
       e.name AS employee_name
     , e.department_id AS department_id
     , COUNT(*) AS duplicates
FROM employees e
GROUP BY e.name, e.department_id
HAVING COUNT(*) > 1;

Task 4:
SELECT
      d.name AS department_name
    , COUNT(e.id) AS employee_count
    , ROUND(AVG(e.salary), 2) AS average_salary
FROM departments d
LEFT JOIN employees e ON d.id = e.department_id
GROUP BY d.name
HAVING COUNT(e.id) > 5 AND AVG(e.salary) > 70000;

Task 5:
SELECT
       e1.name AS employee_name
     , e2.name AS manager_name
FROM employees e1
LEFT JOIN employees e2 ON e1.manager_id = e2.id;
